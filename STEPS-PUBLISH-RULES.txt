🚀 INSTRUCTIONS URGENTES - Publication des Règles Storage
================================================================

❌ PROBLÈME : Upload d'images bloqué (erreur 403)
✅ SOLUTION : Publier les nouvelles règles ci-dessous

📋 ÉTAPES À SUIVRE (2 minutes) :

1. Ouvrez le lien suivant dans votre navigateur :
   https://console.firebase.google.com/project/annonces-app-44d27/storage/rules

2. Vous verrez l'éditeur de règles Storage. Supprimez TOUT le contenu actuel.

3. Copiez-collez le contenu suivant dans l'éditeur :

================================================================
rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    function isAuthenticated() {
      return request.auth != null;
    }
    function isOwner(userId) {
      return isAuthenticated() && request.auth.uid == userId;
    }
    function isAdmin() {
      return isAuthenticated() && request.auth.token.admin == true;
    }
    function isImageContent() {
      let ct = request.resource.contentType.toLowerCase();
      return ct == 'image/jpeg' || ct == 'image/png' || ct == 'image/webp' || ct == 'image/gif' || ct == 'image/jpg';
    }
    function isPdfContent() {
      return request.resource.contentType.toLowerCase() == 'application/pdf';
    }
    function fileExists() {
      return request.resource != null && request.resource.size > 0;
    }
    function isValidSize(maxMB) {
      return request.resource.size < maxMB * 1024 * 1024;
    }

    match /users/{userId}/{fileName} {
      allow read: if true;
      allow write: if isOwner(userId)
                   && isImageContent()
                   && request.resource.size < 5 * 1024 * 1024
                   && fileName.toLowerCase().matches('^profile_[0-9]+\\.(jpg|jpeg|png|webp)$');
      allow delete: if isOwner(userId);
    }

    match /listings/{userId}/{fileName} {
      allow read: if true;
      allow write: if isAuthenticated()
                   && isOwner(userId)
                   && isImageContent()
                   && isValidSize(15)
                   && fileName.toLowerCase().matches('^(image_|photo_)[0-9]+_[0-9]+\\.(jpg|jpeg|png|webp)$');
      allow delete: if isOwner(userId);
    }

    match /verifications/{userId}/{allPaths=**} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated();
      allow update: if false;
      allow delete: if isAuthenticated();
    }

    match /temp/{userId}/{fileName} {
      allow write: if isOwner(userId)
                   && request.resource.size < 50 * 1024 * 1024
                   && request.resource.metadata.keys().hasAny(['temporary']);
      allow read, delete: if isOwner(userId) || isAdmin();
    }

    match /messages/{conversationId}/{fileName} {
      allow read, write, delete: if isAuthenticated();
    }

    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
================================================================

4. Cliquez sur le bouton "Publier" (en haut à droite de l'écran)

5. Attendez 1-2 minutes que Firebase déploie les règles

6. Rechargez votre application et réessayez l'upload d'images

✅ VOUS POURREZ ALORS :
   - Uploader des images lors de la création d'annonces
   - Plus d'erreur 403 Forbidden
